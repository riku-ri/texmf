\gdef\bigmu#1*#2{
	\begingroup\endlinechar=-1\catcode"09=\catcode0\relax%<TAB>,9,'011"09
	\muskip0=#2\relax
	\muskip1=#2\relax
	\count0=#1\advance\count0 by -1000\relax
	\multiply\muskip0 by 1000\relax
	\multiply\muskip1 by \count0\relax
	\advance\muskip0 by \muskip1\relax
	\divide\muskip0 by 1000\relax
	\global#2=\muskip0\relax
	\endgroup
}
\gdef\bigdi#1*#2{
	\begingroup\endlinechar=-1\catcode"09=\catcode0\relax%<TAB>,9,'011"09
	\dimen0=#2\relax
	\dimen1=#2\relax
	\count0=#1\advance\count0 by -1000\relax
	\multiply\dimen0 by 1000\relax
	\multiply\dimen1 by \count0\relax
	\advance\dimen0 by \dimen1\relax
	\divide\dimen0 by 1000\relax
	\global#2=\dimen0\relax
	\endgroup
}
