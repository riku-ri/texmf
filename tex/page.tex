\begingroup\endlinechar=-1%
\expandafter\gdef\csname\string\page\endcsname#1#2#3#4{
\ifx\relax#1\relax
#3=\the#3
\else#3=#1\fi
\ifx\relax#2\relax
#4=\the#4
\else#4=#2\fi
\hsize=\the\pagewidth\vsize=\the\pageheight
\advance\hsize-\pageleftoffset\advance\hsize-\pagerightoffset
\advance\vsize-\pagetopoffset\advance\vsize-\pagebottomoffset
\relax
}
\gdef\pageL#1#2{\csname\string\page\endcsname{#1}{#2}{\pagewidth}{\pageleftoffset}\relax}
\gdef\pageR#1#2{\csname\string\page\endcsname{#1}{#2}{\pagewidth}{\pagerightoffset}\relax}
\gdef\pageLR#1#2{\pageL{#1}{#2}\pageR{#1}{#2}}
\gdef\pageT#1#2{\csname\string\page\endcsname{#1}{#2}{\pageheight}{\pagetopoffset}\relax}
\gdef\pageB#1#2{\csname\string\page\endcsname{#1}{#2}{\pageheight}{\pagebottomoffset}\relax}
\gdef\pageTB#1#2{\pageT{#1}{#2}\pageB{#1}{#2}}
\endgroup%
